Инсталляция пакетов с DVD диска Oracle Linux

1) Инсталляция пакетов с DVD диска Oracle Linux:

# nano /etc/yum.repos.d/public-yum-ol6.repo


[OEL63_DVD]
name=Oracle Enterprise Linux DVD
baseurl=file:///media/"OL6.3 x86_64 Disc 1 20120626"/Server/
gpgcheck=0
enabled=1

Кавычки для пробелов. Все остальные enabled=0

# yum repolist
# yum update -y

Если диск не примонтирован
# mkdir /mnt/cdrom
# mount -t iso9660 -o ro /dev/cdrom /mnt/cdrom


[OEL63_DVD]
name=Oracle Enterprise Linux DVD
baseurl=file:///mnt/cdrom/Server/
gpgcheck=0
enabled=1

2) Инсталляция пакетов из репозитория Oracle Linux в интернете:
# nano /etc/yum.repos.d/public-yum-ol6.repo


[OEL_INTERNET]
name=Oracle Enterprise Linux $releasever - $basearch
baseurl=http://public-yum.oracle.com/repo/OracleLinux/OL6/latest/$basearch/
gpgkey=http://public-yum.oracle.com/RPM-GPG-KEY-oracle-ol6
gpgcheck=1
enabled=1
