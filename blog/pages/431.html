sql интервал времени в секундах

-- перечисление даты/времени с точностью до секунды

select sysdate-level/(24*60*60) daten from dual
connect by level <= (sysdate-(sysdate-24*60*60));

/*
01.12.2013 22:22:41
01.12.2013 22:22:40
01.12.2013 22:22:39
и т.д
*/

select trunc(sysdate+1)-level/(24*60*60) daten from dual
connect by level <= (sysdate-(sysdate-24*60*60))

/*
01.12.2013 23:59:59
01.12.2013 23:59:58
...
01.12.2013 0:00:01
01.12.2013    (0:00:00)

*/
